// OSL -> PEK Fare Watch — offline cache const CACHE = "farewatch-v1"; const ASSETS = [ "./", "./index.html", "./manifest.webmanifest", "./icons/icon-192.png", "./icons/icon-512.png", "./icons/icon-180.png", "./icons/favicon.png" ]; self.addEventListener("install", (e) => { e.waitUntil(caches.open(CACHE).then((c) => c.addAll(ASSETS))); self.skipWaiting(); }); self.addEventListener("activate", (e) => { e.waitUntil( caches.keys().then((keys) => Promise.all(keys.filter((k) => k !== CACHE).map((k) => caches.delete(k))) ) ); self.clients.claim(); }); // cache-first for app shell, fall back to network self.addEventListener("fetch", (e) => { if (e.request.method !== "GET") return; e.respondWith( caches.match(e.request).then((hit) => hit || fetch(e.request)) ); });
